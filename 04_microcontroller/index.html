int current_speed = 0; int previous_speed = 0; void setup() { Serial.begin(9600); delay(500); Serial.println(); Serial.println("--------------------------"); Serial.println("Starting..."); Serial.println(); pinMode(A1A, OUTPUT); // specify these pins as outputs pinMode(A1B, OUTPUT); ledcAttach(A1A, freq, resolution); ledcWrite(A1A, 0); // start with the motors off digitalWrite(A1B, LOW); } void loop() { Serial.println("What should I do now?"); if (previous_speed == 0) { // only if it isn't moving set the default, otherwise just continue current_speed = 200; } else { // Set the current speed to what we last left off current_speed = previous_speed;
